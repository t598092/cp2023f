                            #include <string.h>
                            #include <math.h>

                            #define shapeText(TYPE) char name[10]; \
                                                    int (*perimeter)(struct TYPE*); \
                                                    int (*area)(struct TYPE*);

                            typedef struct shape_s {
                                shapeText(shape_s); 
                            } shape_t;

                            typedef struct circle_s {
                                shapeText(circle_s);
                                int radius;
                            } circle_t;

                            int CirclePerimeter(circle_t *obj) {}
                            int CircleArea(circle_t *obj) {}

                            void CircleNew(circle_t *obj) {
                                obj->perimeter = CirclePerimeter;
                                obj->area = CircleArea;
                            }

                            int main()
                            {
                                circle_t c;
                                CircleNew(&c);
                                int radius;
                                
                                scanf("%d", &radius);
                                c.radius = radius;

                                shape_t *sq = (shape_t*)&c;
                                printf("circle %d %d", sq->perimeter(sq), sq->area(sq));
                            }

Function pointer



  練習使用函式指標 

                        #include <stdio.h>

                        int add(int a, int b);
                        int multiply(int a, int b);

                        int main()
                        {
                            int (*op)(int, int) = add; // op 是函式指標，指向型態為 int (int, int) 的函式
                            int result = op(3, 5); //傳入引數 3, 5 至 op 指向的函式
                            printf("%d ", result);

                            op = multiply; // op 可以指向任意型態為 int (int, int) 的函式
                            result = op(3, 5);
                            printf("%d", result);

                            return 0;
                        }

                        int add(int a, int b)
                        {
                            return a+b;
                        }

                        int multiply(int a, int b)
                        {
                            return a*b
                        }
  output:
  8 15

Function pointer



  練習將函式指標應用於函式參數

                        #include <stdio.h>

                        int calculate(int result, int (*op)(int, int)); //需傳入整數參數與函式指標
                        int add(int a, int b);
                        int multiply(int a, int b);

                        int main()
                        {
                            int result = 1;

                            printf("%d ",calculate(result, add)); //printf("%d ",calculate(result, &add));
                            printf("%d ",calculate(result, multiply)); //printf("%d ",calculate(result, &multiply));

                            return 0;
                        }

                        int add(int a, int b)
                        {
                            return a+b;
                        }

                        int multiply(int a, int b)
                        {
                            return a*b;
                        }

                        int calculate(int result, int (*op)(int, int)) //需傳入整數參數與函式指標
                        {
                            for(int i=2; i<=5; i++)
                            {
                                result = op(result, i);
                            }

                            return result;
                        }
  output:
  15 120
